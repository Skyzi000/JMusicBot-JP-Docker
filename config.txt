//______________________________________________________________
//                     ===================
//                      JMusicBot JPの設定
//                     ===================
//
// //で始まる行は無視されます。
// トークンと所有者を設定しなければなりません。
// 設定しない場合、他のすべての項目にはデフォルトがあります。
// Notepad++などのエディタで編集する事を推奨します。
//￣￣￣￣￣￣￣￣￣￣￣￣￣￣￣￣￣￣￣￣￣￣￣￣￣￣￣￣￣￣￣￣￣￣

// この項目は、Botがログインするためのトークンを設定します。
// 入力するトークンはBotトークンである必要があります（ユーザートークンは機能しません）。
// Botトークンの取得方法がわからない場合は、こちらのガイド（英語）を参照してください:
// https://github.com/jagrosh/MusicBot/wiki/Getting-a-Bot-Token
token = "Botトークンをここに貼り付け"

// この項目は、Botの所有者を設定します。
// 所有者のID（17〜18桁の数字）である必要があります。
// ガイドはこちら（英語）: https://github.com/jagrosh/MusicBot/wiki/Finding-Your-User-ID
owner = 所有者IDをここに貼り付け


// この項目は、Botの接頭辞を設定します。
// 接頭辞はコマンドを制御するために使用されます。
// 例えば、`!!` を使用すると、`play` コマンドは `!!play` になります。
// デフォルトの接頭辞はBotの@mentionになります（例: `@Botname play`）。
prefix = "@mention"

// この項目は、Botの副接頭辞を設定します。
// この項目を設定すると、上で設定した接頭辞に加えて、ここで設定した接頭辞も使用できるようになります。
// `NONE` または `なし` にすると副接頭辞が無効になります。
// デフォルトは `なし` です。
altprefix = "なし"

// この項目を設定すると、Botのゲーム欄に表示される内容を変更できます。
// ゲーム欄は「Playing ～」「Listening to ～」または「Watching ～」に設定できます。
// ※ `～` にはゲーム名を入力してください。
// デフォルトの設定を使用するには `DEFAULT` と入力してください。
// デフォルトは `Playing` です。ゲーム欄に何も表示しない場合は、`NONE` または `なし` と入力してください。
game = "DEFAULT"

// この項目を設定すると、Botのオンライン状態を設定できます。
// 使用できる設定: `ONLINE`（オンライン）、`IDLE`（退席中）、`DND`（取り込み中）、`INVISIBLE`（オンライン状態を隠す）
// デフォルトは `ONLINE` です。
status = ONLINE

// この項目を `true` に設定すると、Botはゲーム欄に再生中の曲を表示します。
// この機能は、Botが1つのサーバーのみで再生している時にのみ動作します。
// （Botが複数のサーバーに参加していても、1サーバーでのみ再生している場合には表示されます。）
// Botが複数のサーバーで再生している場合、この機能は動作しません。
// デフォルトは `true` です。無効にするには `false` に変更してください。
songinstatus = true

// この項目を変更すると、コマンド実行時に表示される絵文字を変更することができます。
success = "🎶"
warning = "💡"
error = "🚫"
loading = "⌚"
searching = "🔎"

// この項目を変更すると、ヘルプメッセージを送信するためのコマンドを設定できます。
// 例えば、接頭辞を `!!` に設定し、この項目を `cmds` に設定すると、
// `!!cmds` と入力することでヘルプメッセージが送信されるようになります。
// デフォルトは `help` です。
help = "help"

// ヘルプメッセージをDMで送信するかどうかを設定できます。`true` に設定すると、DMに送信されます。
// デフォルトは `true` です。
helptodm = true

// この項目を `true` に設定すると、`nowplaying` コマンドで右上に YouTube サムネイルが表示されるようになります。
// 注意: この項目を `true` にすると、`nowplaying` コマンドの自動更新は実行されなくなります。
// （更新時に画像が再度読み込まれてしまうため、自動更新が無効になります。）
// デフォルトは `true` です。無効にするには、`false` に変更してください。
npimages = true

// この項目を `true` に設定すると、ニックネームに Bot の再生状態が表示されるようになります。
// デフォルトは `true` です。無効にするには、`false` に変更してください。
changenickname = true


// ヘルプコマンドに Cosgy Dev 公式 Discord サーバーへの招待リンクを表示します。
// `false` に設定すると、招待リンクは表示されません。
officialinvite = false

// この項目を `true` に設定すると、Botはすべての曲を再生し終えても音声チャンネルから離脱しなくなります。
// 音声チャンネルへの接続を維持することで追加の帯域幅（通信）が必要になるため、通信が不安定になる場合はこのオプションを有効にしないでください。
// デフォルトは `false` です。有効にするには、`true` に変更してください。
stayinchannel = false

// `pausenousers` を `true` に設定すると、Botはボイスチャンネルから全てのユーザーが離脱した際に楽曲の再生を一時停止します。
// ユーザーが再度チャンネルに入ったときに自動で再生を再開したい場合は、`resumejoined` を `true` に設定してください。
// 全員が離脱した時に一時停止ではなく停止させたい場合は、`stopnousers` を `true` に設定してください。
//
// `pausenousers` と `stopnousers` の両方を `true` にした場合、`pausenousers` が優先されます。
//
// `pausenousers` のデフォルトは `false` です。有効にするには、`true` に設定してください。
// `resumejoined` のデフォルトは `false` です。有効にするには、`true` に設定してください。
// `stopnousers` のデフォルトは `false` です。有効にするには、`true` に設定してください。
pausenousers = false
resumejoined = false
stopnousers = false

// ニコニコ動画の再生を有効にする場合、この設定を `true` に変更してください。
// 注意: ニコニコ動画の再生機能はベータ版であり、さまざまな問題が発生する可能性があります。
// デフォルトは `false` です。有効にする場合は `true` に変更してください。
useniconico = false

// useniconicoの設定をtrueにした場合、ニコニコ動画アカウントのメールアドレスとパスワード（ログイン資格情報）を入力してください。
// nicomail: ニコニコ動画アカウントのメールアドレス
// nicopass: ニコニコ動画アカウントのパスワード
nicomail = "example@hoge.jp"
nicopass = "0123456789"

// ニコニコ動画で二段階認証を有効にしている場合、シークレットキーを入力してください。
// 二段階認証を有効にしていない場合は、何も入力せずにそのままにしてください。
// 正しいシークレットキーを入力すると、再生時に自動的に二段階認証コードが生成され、ログインが行われます。
nicotwofactor=""

// YouTubeにログインして使用したい場合は、こちらにログイン情報を入力してください。
// YouTubeの再生はログイン無しでも行えますが、YouTubeプレミアム限定動画などを再生するには、プレミアム登録済みのアカウントが必要です。
// 使用するアカウントが二段階認証を有効にしている場合は、アプリパスワードを使用してください。
ytemail = ""
ytpass = ""

// SpotifyAPIのクライアントキーとシークレットキーを入力してください。
// SpotifyのURLから曲の情報を取得するためにSpotifyAPIを使用します。
spclient = ""
spsecret = ""

// この項目は、1曲の最大再生時間を設定します。
// 0以下に設定すると、無制限になります。
// この制限は、任意の場所から曲が読み込まれた際に適用されます。
// デフォルトは `0` です。
maxtime = 0

// ボイスチャンネルにBotが単独でいる場合、何秒後に退出するかを設定します。
// 退出する際、再生待ちはリセットされます。この設定が `0` 以下、または未設定の場合、この機能は無効になります。
alonetimeuntilstop = 0

// 自動的にボイスチャンネルから切断する際、再生待ちにある曲を保存するかどうかを設定します。
// `true` に設定すると、再生待ちリストが保存されます。デフォルトは `false` です。
autostopqueuesave = false

// この項目は、再生リストの保存先フォルダーを設定します。
// 相対パスまたは絶対パスで入力してください。
// デフォルトは `Playlists` です。
playlistsfolder = Playlists

// この項目は、マイリストの保存先フォルダーを設定します。
// 相対パスまたは絶対パスで入力してください。
// デフォルトは `Mylists` です。

mylistfolder = Mylists

publistfolder = Publist

// この項目は、Botが実行中に新しいバージョンが利用可能になったとき、Botが所有者にDMで通知するかどうかを設定します。
// 新しいバージョンが利用可能な場合に通知するには `true`、通知しない場合は `false` を設定してください。
// デフォルトは `true` です。
updatealerts = true

// この項目は、ユーザーがコマンドを実行したとき、その記録をログに記録するかどうかを設定します。
// コマンドの実行記録を保存する場合は `true`、記録しない場合は `false` と設定してください。
auditcommands = false

// `invite` コマンドを有効にするかを設定します。
// `invite` コマンドを有効にする場合は `true` に設定してください。
// デフォルトは `false` です。
useinvitecommand = false

// この項目は、歌詞を検索するサイトを設定します。
// 利用可能なサイトは次の通りです: `"AZ Lyrics"`, `"Genius"`, `"MusicMatch"`, `"LyricsFreak"`
// 現時点では `"AZ Lyrics"` または `"MusicMatch"` に設定することを推奨します。
// `"Genius"` は曲以外の検索結果を表示する傾向があり、求めている歌詞が表示されない可能性があります。
// プロバイダのコントリビューション（開発）に興味がある場合は、`JLyrics` のリポジトリを参照してください。
// JLyrics リポジトリ: https://github.com/jagrosh/JLyrics
// デフォルトは `"MusicMatch"` です。
lyrics.default = "MusicMatch"

// この項目はコマンドのプレフィックスをカスタマイズできます。
// コンマで区切ることで複数設定することができます。
//
// 例1: "Play"コマンドを"P"でも使えるようにする:
// play = [ p ]
//
// 例2: "search"コマンドを"yts"と"find"で使えるようにする:
// search = [ yts, find ]

aliases {
  // General commands
  help = []
  cache = []
  settings = [status]
  serverinfo = []

  // Music commands
  lyrics = []
  nowplaying = [np, current]
  play = []
  playlists = [pls]
  queue = [list]
  remove = [delete]
  scsearch = []
  ncsearch = [niconicosearch, nicosearch, ncs, nicos]
  search = [ytsearch]
  shuffle = []
  skip = [voteskip]

  // Admin commands
  prefix = [setprefix]
  setdj = []
  setskip = [setskippercent, skippercent, setskipratio]
  settc = []
  setvc = []

  // DJ Commands
  forceremove = [forcedelete, modremove, moddelete, modelete]
  forceskip = [modskip]
  movetrack = [move]
  pause = []
  playnext = []
  repeat = []
  skipto = [jumpto]
  stop = []
  volume = [vol]
  spotify = []
}

// `transforms` は、特定の再生入力を変更し、さまざまな種類の入力に変換するために使用されます。使用例は限られていますが、
// 理論的には、ホワイトリストやブラックリストの設定、特定のソースからの回り道的な読み込み、
// およびカスタマイズされた方法での処理が可能です。
//
// これらは設定が簡単ではないため、使用する場合はコードを確認し、
// それらがどのように機能するかや必要なフィールドを理解する必要があります。
// また、より良い方法が見つかれば、将来的にこの機能が完全に削除される可能性があります。
transforms = {}

// この項目を `true` に設定すると、Botの所有者として設定されたユーザーに対して `eval` コマンドが有効になります。
// このコマンドは、Botのアカウントから任意のコードを実行可能にします。
//
// 警告:
// このコマンドは非常に危険です。
// 完全に理解していない場合は、Discordサーバーやコンピューターに重大な問題を引き起こす可能性があります。
// 実行する内容を完全に理解していない限り、このコマンドを有効にしないでください。
//
// 誰かにこれを有効にするように言われた場合、この機能や使用方法がわからない場合は、有効にしないでください。
// ほとんどの場合、その相手は意図的に危険な操作をさせようとしている可能性があります。
// デフォルトは `false` です。危険性を理解した上で有効にする場合は `true` に変更してください。
eval = false
